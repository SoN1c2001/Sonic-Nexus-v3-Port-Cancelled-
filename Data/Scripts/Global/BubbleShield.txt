sub ObjectDraw
	object.direction = FACING_RIGHT
	if object.timer >= 30
		if object.frame > 1
			object.direction = 2
		end if
	end if
	DrawSpriteFX(object.frame,FX_FLIP,player.xpos,player.ypos)
endsub
