sub ObjectDraw
	if Object.PropertyValue==0
		if Object.State<2
			TempValue0=Object.Frame
			TempValue0/=30
			DrawSpriteFX(TempValue0,FX_FLIP,Object.XPos,Object.YPos)
		else
			DrawSpriteFX(4,FX_FLIP,Object.XPos,Object.YPos)
		endif
	else
		if Object.State<2
			TempValue0=Object.Frame
			TempValue0/=6
			TempValue0+=2
			DrawSpriteFX(TempValue0,FX_FLIP,Object.XPos,Object.YPos)
		else
			DrawSpriteFX(4,FX_FLIP,Object.XPos,Object.YPos)
		endif
	endif
endsub
